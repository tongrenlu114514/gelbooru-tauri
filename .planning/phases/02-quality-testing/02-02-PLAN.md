    ```typescript
    describe('Page State Management', () => {
      it('should save current page state', () => {
        const store = useGalleryStore();
        store.posts = [{ id: 1, url: 'url', title: 'title', thumbnail: null }];
        store.tags = [{ text: 'tag', tagType: 'general', count: 100 }];
        store.currentPage = 5;
        store.totalPages = 10;

        store.savePageState(['tag1'], 'safe');

        expect(store.pageState).not.toBeNull();
        expect(store.pageState?.currentPage).toBe(5);
        expect(store.pageState?.posts).toHaveLength(1);
      });

      it('should restore saved page state', () => {
        const store = useGalleryStore();
        store.savePageState(['tag1'], 'safe');
        // Set current state different from saved
        store.currentPage = 1;

        const restored = store.restorePageState();
        expect(restored).not.toBeNull();
        expect(restored?.currentPage).toBe(1); // from savePageState call above
      });

      it('should return null when no state saved', () => {
        const store = useGalleryStore();
        const restored = store.restorePageState();
        expect(restored).toBeNull();
      });

      it('should clear page state', () => {
        const store = useGalleryStore();
        store.savePageState(['tag1'], 'safe');
        store.clearPageState();
        expect(store.pageState).toBeNull();
      });
    });
    ```

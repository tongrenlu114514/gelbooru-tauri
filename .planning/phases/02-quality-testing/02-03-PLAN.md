    #[cfg(test)]
    mod tests {
        use super::*;
        use tempfile::TempDir;
        use std::fs;

        fn create_test_db() -> (TempDir, Database) {
            let temp_dir = TempDir::new().expect("Failed to create temp dir");
            let db = Database::new(temp_dir.path().to_str().unwrap())
                .expect("Failed to create database");
            (temp_dir, db)
        }

        #[test]
        fn test_add_and_check_favorite_tag() {
            let (_dir, db) = create_test_db();
            let id = db.add_favorite_tag("saber", "character").unwrap();
            assert!(id > 0);
            assert!(db.is_tag_favorited("saber"));
            assert!(!db.is_tag_favorited("rin"));
        }

        #[test]
        fn test_add_favorite_tag_with_parent() {
            let (_dir, db) = create_test_db();
            let parent_id = db.add_favorite_tag("fate_series", "copyright").unwrap();
            let child_id = db.add_favorite_tag_with_parent("saber", "character", parent_id).unwrap();
            assert!(child_id > 0);

            let children = db.get_child_tags(parent_id).unwrap();
            assert_eq!(children.len(), 1);
            assert_eq!(children[0].tag, "saber");
        }

        #[test]
        fn test_remove_favorite_tag() {
            let (_dir, db) = create_test_db();
            let id = db.add_favorite_tag("test_tag", "general").unwrap();
            assert!(db.is_tag_favorited("test_tag"));

            db.remove_favorite_tag(id).unwrap();
            assert!(!db.is_tag_favorited("test_tag"));
        }

        #[test]
        fn test_favorite_post_crud() {
            let (_dir, db) = create_test_db();
            db.add_favorite(12345).unwrap();
            assert!(db.is_favorite(12345));
            assert!(!db.is_favorite(99999));

            db.remove_favorite(12345).unwrap();
            assert!(!db.is_favorite(12345));
        }

        #[test]
        fn test_settings_crud() {
            let (_dir, db) = create_test_db();

            db.set_setting("theme", "dark").unwrap();
            db.set_setting("language", "en").unwrap();

            assert_eq!(db.get_setting("theme").unwrap(), Some("dark".to_string()));
            assert_eq!(db.get_setting("language").unwrap(), Some("en".to_string()));
            assert_eq!(db.get_setting("missing").unwrap(), None);

            let all_settings = db.get_all_settings().unwrap();
            assert_eq!(all_settings.len(), 2);
        }

        #[test]
        fn test_settings_overwrite() {
            let (_dir, db) = create_test_db();

            db.set_setting("theme", "light").unwrap();
            db.set_setting("theme", "dark").unwrap();

            let all = db.get_all_settings().unwrap();
            assert_eq!(all.len(), 1);
            assert_eq!(db.get_setting("theme").unwrap(), Some("dark".to_string()));
        }

        #[test]
        fn test_download_task_persistence() {
            let (_dir, db) = create_test_db();

            let task = DownloadTaskRecord {
                id: 1,
                post_id: 12345,
                file_name: "test.jpg".to_string(),
                file_path: "/downloads/test.jpg".to_string(),
                image_url: "https://example.com/image.jpg".to_string(),
                status: "pending".to_string(),
                progress: 0.0,
                downloaded_size: 0,
                total_size: 1024,
                error_message: None,
            };

            db.save_download_task(&task).unwrap();

            let tasks = db.get_all_download_tasks().unwrap();
            assert_eq!(tasks.len(), 1);
            assert_eq!(tasks[0].post_id, 12345);
            assert_eq!(tasks[0].status, "pending");
        }

        #[test]
        fn test_update_download_task_progress() {
            let (_dir, db) = create_test_db();

            let task = DownloadTaskRecord {
                id: 1,
                post_id: 12345,
                file_name: "test.jpg".to_string(),
                file_path: "/downloads/test.jpg".to_string(),
                image_url: "https://example.com/image.jpg".to_string(),
                status: "downloading".to_string(),
                progress: 0.0,
                downloaded_size: 0,
                total_size: 1024,
                error_message: None,
            };
            db.save_download_task(&task).unwrap();

            db.update_download_task_progress(1, "downloading", 50.0, 512, 1024).unwrap();

            let tasks = db.get_all_download_tasks().unwrap();
            assert_eq!(tasks[0].progress, 50.0);
            assert_eq!(tasks[0].downloaded_size, 512);
        }

        #[test]
        fn test_update_download_task_error() {
            let (_dir, db) = create_test_db();

            let task = DownloadTaskRecord {
                id: 1,
                post_id: 12345,
                file_name: "test.jpg".to_string(),
                file_path: "/downloads/test.jpg".to_string(),
                image_url: "https://example.com/image.jpg".to_string(),
                status: "downloading".to_string(),
                progress: 25.0,
                downloaded_size: 256,
                total_size: 1024,
                error_message: None,
            };
            db.save_download_task(&task).unwrap();

            db.update_download_task_error(1, "Network timeout").unwrap();

            let tasks = db.get_all_download_tasks().unwrap();
            assert_eq!(tasks[0].status, "failed");
            assert_eq!(tasks[0].error_message, Some("Network timeout".to_string()));
        }

        #[test]
        fn test_delete_download_task() {
            let (_dir, db) = create_test_db();

            let task = DownloadTaskRecord {
                id: 1,
                post_id: 12345,
                file_name: "test.jpg".to_string(),
                file_path: "/downloads/test.jpg".to_string(),
                image_url: "https://example.com/image.jpg".to_string(),
                status: "pending".to_string(),
                progress: 0.0,
                downloaded_size: 0,
                total_size: 1024,
                error_message: None,
            };
            db.save_download_task(&task).unwrap();

            db.delete_download_task(1).unwrap();

            let tasks = db.get_all_download_tasks().unwrap();
            assert!(tasks.is_empty());
        }

        #[test]
        fn test_get_all_favorite_tags_empty() {
            let (_dir, db) = create_test_db();
            let tags = db.get_all_favorite_tags().unwrap();
            assert!(tags.is_empty());
        }
    }

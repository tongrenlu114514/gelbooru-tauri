<task type="auto">
  <name>Task 1: Install Husky and lint-staged</name>
  <files>package.json</files>
  <action>
    1. Install husky and lint-staged as dev dependencies:
    ```bash
    pnpm add -D husky lint-staged
    ```

    2. Initialize Husky:
    ```bash
    npx husky init
    ```

    This creates .husky/ directory with pre-commit script.

    3. Update .husky/pre-commit to run lint-staged:
    ```bash
    echo 'npx lint-staged' > .husky/pre-commit
    ```

    4. Add lint-staged configuration to package.json:
    ```json
    "lint-staged": {
      "src/**/*.{ts,vue}": ["eslint --fix", "prettier --write"],
      "src-tauri/src/**/*.rs": "cd src-tauri && cargo fmt && cargo clippy -- -D warnings && cd .."
    },
    ```

    5. Add prepare script if not present:
    ```json
    "prepare": "husky"
    ```
  </action>
  <verify>
    <automated>ls -la .husky/pre-commit && cat .husky/pre-commit</automated>
  </verify>

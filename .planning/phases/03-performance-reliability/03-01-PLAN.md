<objective>
Fix the imageCache memory leak in Gallery.vue using IntersectionObserver-based lazy loading. Replace the current `preloadImages` function that iterates all images and converts them to base64, with a viewport-aware loading pattern that only loads visible images and disconnects the observer when the component unmounts.
</objective>

<context>
@src/views/Gallery.vue
@src/utils/lruCache.ts
@src/tests/setup.ts

The existing `preloadImages` function (lines 47-58 in Gallery.vue) iterates ALL image paths and converts each to base64, storing in `imageBase64Cache`. This is the memory leak source: with 1000+ images, 100 * ~500KB per base64 = 50MB+ of base64 strings all kept in the LRU cache.

The fix: use `convertFileSrc` for the `src` attribute (which bypasses the base64 cache entirely), and use IntersectionObserver to trigger base64 conversion only for visible images. When images leave the viewport, they are evicted naturally by the LRU cache (max 100 entries).
</context>

2. **Add `loadVisibleImages` function** (replaces `preloadImages`):
   ```typescript
   function loadVisibleImages() {
     const grid = document.querySelector('.content-grid');
     if (!grid || !observerRef.value) return;
     const cards = grid.querySelectorAll<HTMLElement>('[data-image-path]');
     cards.forEach((card) => observerRef.value!.observe(card));
   }
   ```

3. **Add `observeCallback` function** for IntersectionObserver entries:
   ```typescript
   function observeCallback(entries: IntersectionObserverEntry[]) {
     entries.forEach((entry) => {
       if (entry.isIntersecting) {
         const path = (entry.target as HTMLElement).dataset.imagePath;
         if (path && !imageBase64Cache.has(path)) {
           loadImageBase64(path);
         }
       }
     });
   }
   ```

4. **Add `loadImageBase64` helper** (refactored from `handleImageError`):
   ```typescript
   async function loadImageBase64(path: string) {
     if (imageBase64Cache.has(path)) return;
     try {
       const base64 = await invoke<string>('get_local_image_base64', { path });
       imageBase64Cache.set(path, base64);
     } catch (err) {
       console.warn('Failed to load image base64:', path, err);
     }
   }
   ```

5. **Create observer in `onMounted`** (replace/extend existing):
   ```typescript
   onMounted(() => {
     loadTree();
     window.addEventListener('keydown', handleKeydown);
     observerRef.value = new IntersectionObserver(observeCallback, {
       root: null,
       rootMargin: '200px',
       threshold: 0.01,
     });
   });
   ```

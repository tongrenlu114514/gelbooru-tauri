3. **Add `download_with_retry` function** (before the `#[tauri::command]` functions or as a standalone async fn):
   ```rust
   /// Download with exponential backoff retry.
   /// fetch: closure that performs one download attempt
   /// cancel_rx: cancellation channel
   async fn download_with_retry<F, Fut>(
       mut fetch: F,
       mut cancel_rx: mpsc::Receiver<()>,
   ) -> Result<reqwest::Response, String>
   where
       F: FnMut() -> Fut,
       Fut: std::future::Future<Output = Result<reqwest::Response, reqwest::Error>>,
   {
       let mut attempt: u32 = 1;

       loop {
           // Race: download attempt vs cancellation
           let response = tokio::select! {
               result = fetch() => {
                   match result {
                       Ok(resp) => resp,
                       Err(e) => {
                           // Transport error (timeout, connect, builder) — retryable
                           if attempt >= MAX_RETRIES {
                               return Err(format!("Request failed after {} attempt(s): {}", attempt, e));
                           }
                           // Exponential backoff: 1s, 2s, 4s
                           let delay = tokio::time::Duration::from_millis(
                               BASE_DELAY_MS * (2_u64.pow(attempt.saturating_sub(1)))
                           );
                           tokio::select! {
                               _ = tokio::time::sleep(delay) => {
                                   attempt += 1;
                                   continue;
                               }
                               _ = cancel_rx.recv() => {
                                   return Err("cancelled".to_string());
                               }
                           }
                       }
                   }
               }
               _ = cancel_rx.recv() => {
                   return Err("cancelled".to_string());
               }
           };

           // Check HTTP status on Ok response
           if !response.status().is_server_error() {
               // 2xx, 4xx, etc. — do not retry
               return Ok(response);
           }

           // 5xx — retryable
           if attempt >= MAX_RETRIES {
               return Err(format!(
                   "Server error {} after {} retries",
                   response.status(),
                   MAX_RETRIES
               ));
           }

           let delay = tokio::time::Duration::from_millis(
               BASE_DELAY_MS * (2_u64.pow(attempt.saturating_sub(1)))
           );
           attempt += 1;

           tokio::select! {
               _ = tokio::time::sleep(delay) => {
                   // continue loop
               }
               _ = cancel_rx.recv() => {
                   return Err("cancelled".to_string());
               }
           }
       }
   }
   ```

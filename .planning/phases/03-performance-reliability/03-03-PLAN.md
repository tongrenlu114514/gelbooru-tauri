   ```rust
   /// Scan a single directory asynchronously, then spawn children in parallel.
   /// Semaphore limits concurrent directory handles to prevent FD exhaustion.
   async fn scan_dir(
       dir: std::path::PathBuf,
       sem: std::sync::Arc<tokio::sync::Semaphore>,
   ) -> Option<TreeNode> {
       // Collect child futures WITHOUT holding semaphore (avoids deadlock)
       let child_futures: Vec<_> = {
           let mut entries = match tokio::fs::read_dir(&dir).await.ok()? else { return None; };
           let mut subdirs: Vec<std::path::PathBuf> = Vec::new();
           let mut image_count: usize = 0;
           let mut first_image: Option<String> = None;

           while let Some(entry) = entries.next_entry().await.ok().flatten() {
               let path = entry.path();
               if path.is_dir().await {
                   subdirs.push(path);
               } else if is_image(&path) {
                   image_count += 1;
                   if first_image.is_none() {
                       first_image = Some(path.to_string_lossy().to_string());
                   }
               }
           }

           subdirs
               .into_iter()
               .map(|subdir| {
                   let sem = sem.clone();
                   tokio::spawn(async move { scan_dir(subdir, sem).await })
               })
               .collect()
       };

       // Do NOT hold semaphore across await points
       // Now scan own directory entries
       let mut entries = match tokio::fs::read_dir(&dir).await.ok()? else { return None; };
       let mut children: Vec<TreeNode> = Vec::new();
       let mut image_count: usize = 0;
       let mut first_image: Option<String> = None;

       while let Some(entry) = entries.next_entry().await.ok().flatten() {
           let path = entry.path();
           if path.is_dir().await {
               // Already collected above, skip here
           } else if is_image(&path) {
               image_count += 1;
               if first_image.is_none() {
                   first_image = Some(path.to_string_lossy().to_string());
               }
           }
       }

       // Collect child results
       for future in child_futures {
           if let Ok(Some(child)) = future.await {
               image_count += child.image_count;
               if first_image.is_none() {
                   first_image = child.thumbnail.clone();
               }
               children.push(child);
           }
       }

       if image_count == 0 { return None; }

       let dir_name = dir.file_name()
           .map(|n| n.to_string_lossy().to_string())
           .unwrap_or_else(|| dir.to_string_lossy().to_string());

       Some(TreeNode {
           key: dir.to_string_lossy().to_string(),
           label: dir_name,
           path: dir.to_string_lossy().to_string(),
           is_leaf: children.is_empty(),
           image_count,
           children: if children.is_empty() { None } else { Some(children) },
           thumbnail: first_image,
       })
   }
   ```

   /// Build directory tree asynchronously for a root path.
   /// Scans top-level subdirectories in parallel, each with bounded concurrency.
   async fn build_tree_async(root: std::path::PathBuf) -> Vec<TreeNode> {
       let sem = std::sync::Arc::new(tokio::sync::Semaphore::new(MAX_CONCURRENT_DIRS));

       let mut root_nodes: Vec<TreeNode> = Vec::new();
       let mut subdirs: Vec<std::path::PathBuf> = Vec::new();

       let mut entries = match tokio::fs::read_dir(&root).await {
           Ok(e) => e,
           Err(_) => return Vec::new(),
       };

       while let Some(entry) = entries.next_entry().await.ok().flatten() {
           if entry.path().is_dir().await {
               subdirs.push(entry.path());
           }
       }

       let futures: Vec<_> = subdirs
           .into_iter()
           .map(|subdir| {
               let sem = sem.clone();
               tokio::spawn(async move { scan_dir(subdir, sem).await })
           })
           .collect();

       for future in futures {
           if let Ok(Some(node)) = future.await {
               root_nodes.push(node);
           }
       }

       root_nodes.sort_by_key(|n| n.label.clone());
       root_nodes
   }
   ```

   async fn get_directory_images_async(dir_path: std::path::PathBuf) -> DirectoryImages {
       // Similar pattern: use tokio::fs for reading directory entries
       // and tokio::spawn for parallel subdirectory counting
       // (This can reuse the count_images_in_dir logic but async)
       let mut subdirs: Vec<SubDirInfo> = Vec::new();
       let mut images_with_time: Vec<(String, std::time::SystemTime)> = Vec::new();

       let mut entries = match tokio::fs::read_dir(&dir_path).await {
           Ok(e) => e,
           Err(_) => return DirectoryImages { subdirs, images: Vec::new(), total: 0 },
       };

       while let Some(entry) = entries.next_entry().await.ok().flatten() {
           let path = entry.path();
           let metadata = match tokio::fs::metadata(&path).await.ok() {
               Some(m) => m,
               None => continue,
           };

           if metadata.is_dir().await {
               // Count images in subdir asynchronously
               let (count, first) = count_images_async(path.clone()).await;
               if count > 0 {
                   let dir_name = path.file_name()
                       .map(|n| n.to_string_lossy().to_string())
                       .unwrap_or_else(|| "未命名".to_string());
                   subdirs.push(SubDirInfo {
                       path: path.to_string_lossy().to_string(),
                       name: dir_name,
                       image_count: count,
                       thumbnail: first,
                   });
               }
           } else if is_image(&path) {
               let mtime = metadata.modified().ok()
                   .unwrap_or(std::time::SystemTime::UNIX_EPOCH);
               images_with_time.push((path.to_string_lossy().to_string(), mtime));
           }
       }

       images_with_time.sort_by(|a, b| b.1.cmp(&a.1));
       let images: Vec<ImageInfo> = images_with_time
           .into_iter()
           .map(|(path, _)| {
               let name = path.rsplit(['/', '\\']).next().unwrap_or(&path).to_string();
               ImageInfo { path, name }
           })
           .collect();

       DirectoryImages {
           subdirs,
           images,
           total: images.len(),
       }
   }

   async fn count_images_async(dir: std::path::PathBuf) -> (usize, Option<String>) {
       let sem = std::sync::Arc::new(tokio::sync::Semaphore::new(MAX_CONCURRENT_DIRS));
       let result = count_dir_images(dir, sem).await;
       result
   }

   async fn count_dir_images(dir: std::path::PathBuf, sem: std::sync::Arc<tokio::sync::Semaphore>) -> (usize, Option<String>) {
       let _permit = sem.acquire().await.ok();
       let mut count = 0;
       let mut first: Option<String> = None;

       let mut entries = match tokio::fs::read_dir(&dir).await.ok() {
           Some(e) => e,
           None => return (0, None),
       };

       while let Some(entry) = entries.next_entry().await.ok().flatten() {
           let path = entry.path();
           if path.is_dir().await {
               let (sub_count, sub_first) = count_dir_images(path, sem.clone()).await;
               count += sub_count;
               if first.is_none() { first = sub_first; }
           } else if is_image(&path) {
               count += 1;
               if first.is_none() {
                   first = Some(path.to_string_lossy().to_string());
               }
           }
       }

       (count, first)
   }
   ```

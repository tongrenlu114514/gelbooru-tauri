<script setup lang="ts">
import { ref, onMounted } from 'vue';
import { NSkeleton, NIcon } from 'naive-ui';
import { FolderOutline } from '@vicons/ionicons5';
import { convertFileSrc } from '@tauri-apps/api/core';
import { imageBase64Cache } from '@/utils/lruCache';

interface ImageInfo { path: string; name: string }
interface SubDirInfo { path: string; name: string; imageCount: number; thumbnail?: string }

const props = defineProps<{
  images: ImageInfo[];
  subdirs: SubDirInfo[];
  loadingImages: boolean;
  selectedKey: string | null;
}>();

const emit = defineEmits<{
  'open-preview': [index: number];
  'enter-subdir': [subdir: SubDirInfo];
}>();

// Skeleton count: estimate from viewport (3 rows x auto columns)
const skeletonCount = ref(12);

function getImageSrc(path: string): string {
  if (!path) return '';
  return convertFileSrc(path.replace(/\\/g, '/'));
}

function handleImageError(event: Event, path: string) {
  const img = event.target as HTMLImageElement;
  if (img && path && !imageBase64Cache.has(path)) {
    import('@tauri-apps/api/core').then(({ invoke }) => {
      invoke<string>('get_local_image_base64', { path })
        .then((base64) => {
          imageBase64Cache.set(path, base64);
          img.src = base64;
        })
        .catch(() => {});
    });
  }
}

function updateSkeletonCount() {
  const grid = document.querySelector('.content-grid');
  if (grid) {
    const cellWidth = 164; // 160px + 4px gap
    const cols = Math.max(1, Math.floor(grid.clientWidth / cellWidth));
    skeletonCount.value = Math.max(cols * 3, 6);
  }
}

onMounted(() => {
  const ro = new ResizeObserver(updateSkeletonCount);
  const grid = document.querySelector('.content-grid');
  if (grid) ro.observe(grid);
  updateSkeletonCount();
});
</script>

<template>

    <div v-for="i in skeletonCount" :key="i" class="skeleton-card">
      <n-skeleton :height="160" width="100%" :sharp="false" />
    </div>
  </div>

  <!-- Per D-03, D-04, D-06, D-07, D-11: Apple Photos card grid -->

    <!-- Per D-11: Folder cards unified with image cards -->
    <div
      v-for="subdir in subdirs"
      :key="subdir.path"
      class="gallery-card folder-card"
      :data-image-path="subdir.thumbnail"
      @click="emit('enter-subdir', subdir)"
    >
      <div class="folder-preview">
        <img
          v-if="subdir.thumbnail"
          :src="getImageSrc(subdir.thumbnail)"
          alt=""
          loading="lazy"
          @error="handleImageError($event, subdir.thumbnail)"

      </div>
      <!-- Per D-06: filename only on hover, not visible by default -->
      <!-- Per D-07: no action buttons, click opens preview -->

    </div>

    <div

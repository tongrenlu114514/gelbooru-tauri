```typescript
function handleAutocompleteSearch(tags: string[]) {
  // Trigger search with current tags
  // Similar to existing searchPosts logic but uses selectedTags directly
  galleryStore.currentPage = 1;
  performSearch(tags);
}

// Extract shared search logic
async function performSearch(tags: string[]) {
  const tagsWithRating = [...tags];
  if (selectedRating.value) {
    tagsWithRating.push(selectedRating.value);
  }
  loading.value = true;
  try {
    const result = await invoke<{ postList: GelbooruPost[]; tagList: GelbooruTag[]; totalPages: number }>('search_posts', {
      tags: tagsWithRating,
      page: galleryStore.currentPage,
    });
    galleryStore.setPosts(result.postList);
    galleryStore.setTags(result.tagList);
    galleryStore.setTotalPages(result.totalPages);
    galleryStore.setSearchTags(tagsWithRating);
  } catch (error) {
    console.error('Search failed:', error);
  } finally {
    loading.value = false;
    scrollToTop();
  }
}
```

  <action>
Add a dedicated `retryDownload(id)` action to the download store that resets the task's progress to 0 before calling `startDownload`. This gives immediate visual feedback (progress bar clears) and also adds a `showFailedOnly` ref with a corresponding `filteredTasks` computed that returns only failed tasks when the filter is active. This keeps filter state in the store rather than the view.

```typescript
// Add after failed computed:
const showFailedOnly = ref(false);

const filteredTasks = computed(() => {
  if (showFailedOnly.value) {
    return tasks.value.filter((t) => t.status === 'failed');
  }
  return [...tasks.value].sort((a, b) => b.id - a.id);
});

// Add near other actions:
async function retryDownload(id: number) {
  const index = tasks.value.findIndex((t) => t.id === id);
  if (index !== -1) {
    // Reset progress immediately for instant visual feedback
    tasks.value[index] = { ...tasks.value[index], progress: 0, error: undefined };
  }
  await startDownload(id);
}
```
